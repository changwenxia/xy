    # 隐藏邮箱前两位后的四位

    隐藏邮箱前两位后的四位。

    # 示例：

    12****3434@163.com

    # 代码实现：

    ...
<div>{{mail | hideMailMiddle}}</div>
...  

<script>
import { hideMailMiddle } from "@xiyun/utils";

export default {
    filters: { hideMailMiddle },
    data() {
        return {
            mail: '1234323434@163.com',
        }
    },
}
</script>
